    def initialize(druid, base='.')
      druid = druid.to_s unless druid.is_a? String
      unless self.class.valid?(druid)
        raise ArgumentError, "Invalid DRUID: '#{druid}'"
      end
      druid = [self.class.prefix,druid].join(':') unless druid =~ /^#{self.class.prefix}:/
      @base = base
      @druid = druid
    end
